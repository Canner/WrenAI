# T10 首存用户日龄趋势

## 模板用途

统计首存 cohort 从首日开始的 D1~DN 投充比/杀率趋势。

## 建议问题（可转为 sql_pair.question）

- 统计首存 cohort 从首日开始的 D1~DN 投充比/杀率趋势
- 统计某渠道首存 cohort 在首存后 N 天内的日龄趋势指标

## 核心表/模型

- dwd_order_deposit
- dwd_order_withdrawal
- dwd_bet_order

## 参数

- tenant_plat_id
- channel_id
- cohort_start_date
- cohort_end_date
- n_days

## SQL 模板

```sql
WITH RECURSIVE seq AS (
    SELECT 1 AS relative_day_no
    UNION ALL
    SELECT relative_day_no + 1
    FROM seq
    WHERE relative_day_no < :n_days
),
first_deposit_cohort AS (
    SELECT
        d.tenant_plat_id,
        d.channel_id,
        d.player_id,
        DATE(MIN(d.callback_time)) AS first_deposit_date
    FROM dwd_order_deposit d
    WHERE d.status = 2
      AND d.times = 1
      AND d.tenant_plat_id = :tenant_plat_id
      AND d.channel_id = :channel_id
      AND d.callback_time >= :cohort_start_date
      AND d.callback_time < DATE_ADD(:cohort_end_date, INTERVAL 1 DAY)
    GROUP BY d.tenant_plat_id, d.channel_id, d.player_id
),
cohort_size AS (
    SELECT
        tenant_plat_id,
        channel_id,
        first_deposit_date,
        COUNT(*) AS cohort_user_count
    FROM first_deposit_cohort
    GROUP BY tenant_plat_id, channel_id, first_deposit_date
),
deposit_daily AS (
    SELECT
        c.player_id,
        DATE(d.callback_time) AS event_date,
        SUM(d.actual_amount) AS deposit_amount
    FROM first_deposit_cohort c
    INNER JOIN dwd_order_deposit d
            ON d.player_id = c.player_id
           AND d.tenant_plat_id = c.tenant_plat_id
           AND d.channel_id = c.channel_id
    WHERE d.status = 2
      AND d.callback_time >= c.first_deposit_date
      AND d.callback_time < DATE_ADD(c.first_deposit_date, INTERVAL :n_days DAY)
    GROUP BY c.player_id, DATE(d.callback_time)
),
withdraw_daily AS (
    SELECT
        c.player_id,
        DATE(w.callback_time) AS event_date,
        SUM(w.act_amount) AS withdrawal_amount
    FROM first_deposit_cohort c
    INNER JOIN dwd_order_withdrawal w
            ON w.player_id = c.player_id
           AND w.tenant_plat_id = c.tenant_plat_id
           AND w.channel_id = c.channel_id
    WHERE w.status = 3
      AND w.callback_time >= c.first_deposit_date
      AND w.callback_time < DATE_ADD(c.first_deposit_date, INTERVAL :n_days DAY)
    GROUP BY c.player_id, DATE(w.callback_time)
),
bet_daily AS (
    SELECT
        c.player_id,
        DATE(b.settle_time) AS event_date,
        SUM(b.valid_bet_amount) AS valid_bet_amount,
        SUM(b.win_loss_amount) AS win_loss_amount
    FROM first_deposit_cohort c
    INNER JOIN dwd_bet_order b
            ON b.player_id = c.player_id
           AND b.tenant_plat_id = c.tenant_plat_id
           AND b.channel_id = c.channel_id
    WHERE b.settle_status = 1
      AND b.settle_time >= c.first_deposit_date
      AND b.settle_time < DATE_ADD(c.first_deposit_date, INTERVAL :n_days DAY)
    GROUP BY c.player_id, DATE(b.settle_time)
),
player_day_keys AS (
    SELECT player_id, event_date FROM deposit_daily
    UNION
    SELECT player_id, event_date FROM withdraw_daily
    UNION
    SELECT player_id, event_date FROM bet_daily
),
player_day_metrics AS (
    SELECT
        c.tenant_plat_id,
        c.channel_id,
        c.first_deposit_date,
        DATEDIFF(k.event_date, c.first_deposit_date) + 1 AS relative_day_no,
        k.player_id,
        d.deposit_amount,
        w.withdrawal_amount,
        b.valid_bet_amount,
        b.win_loss_amount
    FROM first_deposit_cohort c
    INNER JOIN player_day_keys k
            ON k.player_id = c.player_id
    LEFT JOIN deposit_daily d
           ON d.player_id = k.player_id
          AND d.event_date = k.event_date
    LEFT JOIN withdraw_daily w
           ON w.player_id = k.player_id
          AND w.event_date = k.event_date
    LEFT JOIN bet_daily b
           ON b.player_id = k.player_id
          AND b.event_date = k.event_date
    WHERE DATEDIFF(k.event_date, c.first_deposit_date) + 1 BETWEEN 1 AND :n_days
),
daily_agg AS (
    SELECT
        tenant_plat_id,
        channel_id,
        first_deposit_date,
        relative_day_no,
        SUM(CASE WHEN deposit_amount IS NOT NULL THEN 1 ELSE 0 END) AS deposit_user_count,
        SUM(COALESCE(deposit_amount, 0)) AS deposit_amount,
        SUM(CASE WHEN withdrawal_amount IS NOT NULL THEN 1 ELSE 0 END) AS withdrawal_user_count,
        SUM(COALESCE(withdrawal_amount, 0)) AS withdrawal_amount,
        SUM(CASE WHEN valid_bet_amount IS NOT NULL THEN 1 ELSE 0 END) AS bet_user_count,
        SUM(COALESCE(valid_bet_amount, 0)) AS valid_bet_amount,
        SUM(COALESCE(win_loss_amount, 0)) AS win_loss_amount
    FROM player_day_metrics
    GROUP BY tenant_plat_id, channel_id, first_deposit_date, relative_day_no
)
SELECT
    cs.tenant_plat_id,
    cs.channel_id,
    cs.first_deposit_date,
    CONCAT('D', s.relative_day_no) AS day_label,
    s.relative_day_no,
    cs.cohort_user_count,
    COALESCE(a.deposit_user_count, 0) AS deposit_user_count,
    COALESCE(a.deposit_amount, 0) AS deposit_amount,
    COALESCE(a.withdrawal_user_count, 0) AS withdrawal_user_count,
    COALESCE(a.withdrawal_amount, 0) AS withdrawal_amount,
    COALESCE(a.deposit_amount, 0) - COALESCE(a.withdrawal_amount, 0) AS charge_withdraw_diff,
    COALESCE(a.bet_user_count, 0) AS bet_user_count,
    COALESCE(a.valid_bet_amount, 0) AS valid_bet_amount,
    COALESCE(a.win_loss_amount, 0) AS win_loss_amount,
    ROUND(COALESCE(a.win_loss_amount, 0) / NULLIF(COALESCE(a.valid_bet_amount, 0), 0), 6) AS kill_rate,
    ROUND(COALESCE(a.valid_bet_amount, 0) / NULLIF(COALESCE(a.deposit_amount, 0), 0), 6) AS bet_deposit_ratio
FROM cohort_size cs
CROSS JOIN seq s
LEFT JOIN daily_agg a
       ON a.tenant_plat_id = cs.tenant_plat_id
      AND a.channel_id = cs.channel_id
      AND a.first_deposit_date = cs.first_deposit_date
      AND a.relative_day_no = s.relative_day_no
ORDER BY cs.first_deposit_date, s.relative_day_no;
```

## 备注

- `D1` 表示首存当日，`D2` 表示首存次日，以此类推。
- 统计对象仅限 `cohort_start_date ~ cohort_end_date` 内的首存用户，但这些用户的后续行为可延伸到 cohort 窗口之外。
- 若运行环境不支持递归 CTE，可改成数字维表/日期维表实现。
- SQL 按 TiDB / MySQL 8 风格编写；导入前需在实际 runtime datasource 下做一次校验。
- 当前可视为 SQL 草案，校验通过后可转为 sql_pair。
