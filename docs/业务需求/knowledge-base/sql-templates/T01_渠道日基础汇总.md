# T01 渠道日基础汇总

## 模板用途

统计某站点/渠道在指定日期范围内每日的登录、注册、充值、提现、投注、优惠、返水、任务彩金。

## 建议问题（可转为 sql_pair.question）

- 统计某站点/渠道在指定日期范围内每日的登录、注册、充值、提现、投注、优惠、返水、任务彩金。
- 按天查看某渠道综合日报指标
- 统计某渠道最近7天的登录、注册、充值、提现、投注汇总

## 核心表/模型

- dwd_player_login_log
- dim_player
- dwd_order_deposit
- dwd_order_withdrawal
- dwd_bet_order
- dwd_order_rebate
- dwd_order_add_or_sub
- dwd_order_activity
- dwd_order_task
- dwd_order_promote_activity
- dwd_order_lottery
- dwd_order_vip_award
- channel

## 参数

- tenant_plat_id
- channel_id
- start_date
- end_date

## SQL 模板

```sql
WITH login_daily AS (
    SELECT
        DATE(l.create_time) AS biz_date,
        l.channel_id,
        COUNT(DISTINCT l.player_id) AS login_user_count
    FROM dwd_player_login_log l
    WHERE l.category = 1
      AND l.tenant_plat_id = :tenant_plat_id
      AND l.channel_id = :channel_id
      AND l.create_time >= :start_date
      AND l.create_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(l.create_time), l.channel_id
),
register_daily AS (
    SELECT
        DATE(p.create_time) AS biz_date,
        p.channel_id,
        COUNT(*) AS register_user_count
    FROM dim_player p
    WHERE p.tenant_plat_id = :tenant_plat_id
      AND p.channel_id = :channel_id
      AND p.create_time >= :start_date
      AND p.create_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(p.create_time), p.channel_id
),
deposit_daily AS (
    SELECT
        DATE(d.callback_time) AS biz_date,
        d.channel_id,
        COUNT(DISTINCT d.player_id) AS deposit_user_count,
        SUM(d.actual_amount) AS deposit_amount,
        COUNT(DISTINCT CASE WHEN d.times = 1 THEN d.player_id END) AS first_deposit_user_count,
        SUM(CASE WHEN d.times = 1 THEN d.actual_amount ELSE 0 END) AS first_deposit_amount,
        SUM(
            CASE
                WHEN d.regist_time >= :start_date
                 AND d.regist_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
                THEN d.actual_amount
                ELSE 0
            END
        ) AS new_customer_deposit_amount
    FROM dwd_order_deposit d
    WHERE d.status = 2
      AND d.tenant_plat_id = :tenant_plat_id
      AND d.channel_id = :channel_id
      AND d.callback_time >= :start_date
      AND d.callback_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(d.callback_time), d.channel_id
),
first_deposit_split AS (
    SELECT
        DATE(d.callback_time) AS biz_date,
        d.channel_id,
        COUNT(
            DISTINCT CASE
                WHEN d.times = 1 AND DATE(d.regist_time) = DATE(d.callback_time) THEN d.player_id
            END
        ) AS new_customer_first_deposit_user_count,
        COUNT(
            DISTINCT CASE
                WHEN d.times = 1 AND DATE(d.regist_time) <> DATE(d.callback_time) THEN d.player_id
            END
        ) AS develop_user_count
    FROM dwd_order_deposit d
    WHERE d.status = 2
      AND d.tenant_plat_id = :tenant_plat_id
      AND d.channel_id = :channel_id
      AND d.callback_time >= :start_date
      AND d.callback_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(d.callback_time), d.channel_id
),
withdraw_daily AS (
    SELECT
        DATE(w.callback_time) AS biz_date,
        w.channel_id,
        COUNT(DISTINCT w.player_id) AS withdrawal_user_count,
        SUM(w.act_amount) AS withdrawal_amount
    FROM dwd_order_withdrawal w
    WHERE w.status = 3
      AND w.tenant_plat_id = :tenant_plat_id
      AND w.channel_id = :channel_id
      AND w.callback_time >= :start_date
      AND w.callback_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(w.callback_time), w.channel_id
),
bet_daily AS (
    SELECT
        DATE(b.settle_time) AS biz_date,
        b.channel_id,
        COUNT(DISTINCT b.player_id) AS bet_user_count,
        SUM(b.valid_bet_amount) AS valid_bet_amount,
        SUM(b.win_loss_amount) AS win_loss_amount
    FROM dwd_bet_order b
    WHERE b.settle_status = 1
      AND b.tenant_plat_id = :tenant_plat_id
      AND b.channel_id = :channel_id
      AND b.settle_time >= :start_date
      AND b.settle_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(b.settle_time), b.channel_id
),
rebate_daily AS (
    SELECT
        DATE(r.receive_time) AS biz_date,
        r.channel_id,
        SUM(r.amount) AS rebate_amount
    FROM dwd_order_rebate r
    WHERE r.status = 1
      AND r.tenant_plat_id = :tenant_plat_id
      AND r.channel_id = :channel_id
      AND r.receive_time >= :start_date
      AND r.receive_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(r.receive_time), r.channel_id
),
add_sub_daily AS (
    SELECT
        DATE(a.modify_time) AS biz_date,
        a.channel_id,
        SUM(CASE WHEN a.add_or_sub_type_id IN (1207, 1209) THEN a.amount ELSE 0 END)
      - SUM(CASE WHEN a.add_or_sub_type_id IN (2204, 2207) THEN a.amount ELSE 0 END) AS discount_adjust_amount
    FROM dwd_order_add_or_sub a
    WHERE a.status = 2
      AND a.add_or_sub_type_id IN (1207, 1209, 2204, 2207)
      AND a.tenant_plat_id = :tenant_plat_id
      AND a.channel_id = :channel_id
      AND a.modify_time >= :start_date
      AND a.modify_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(a.modify_time), a.channel_id
),
vip_award_daily AS (
    SELECT
        DATE(v.modify_time) AS biz_date,
        v.channel_id,
        SUM(v.amount) AS vip_award_amount
    FROM dwd_order_vip_award v
    WHERE v.status = 2
      AND v.tenant_plat_id = :tenant_plat_id
      AND v.channel_id = :channel_id
      AND v.modify_time >= :start_date
      AND v.modify_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(v.modify_time), v.channel_id
),
activity_daily AS (
    SELECT
        DATE(a.receive_time) AS biz_date,
        a.channel_id,
        SUM(a.amount) AS activity_amount
    FROM dwd_order_activity a
    WHERE a.status = 2
      AND a.tenant_plat_id = :tenant_plat_id
      AND a.channel_id = :channel_id
      AND a.receive_time >= :start_date
      AND a.receive_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(a.receive_time), a.channel_id
),
task_daily AS (
    SELECT
        DATE(t.receive_time) AS biz_date,
        t.channel_id,
        SUM(t.amount) AS task_amount
    FROM dwd_order_task t
    WHERE t.status = 2
      AND t.tenant_plat_id = :tenant_plat_id
      AND t.channel_id = :channel_id
      AND t.receive_time >= :start_date
      AND t.receive_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(t.receive_time), t.channel_id
),
promote_daily AS (
    SELECT
        DATE(p.send_time) AS biz_date,
        p.channel_id,
        SUM(p.amount) AS promote_activity_amount
    FROM dwd_order_promote_activity p
    WHERE p.status = 1
      AND p.tenant_plat_id = :tenant_plat_id
      AND p.channel_id = :channel_id
      AND p.send_time >= :start_date
      AND p.send_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(p.send_time), p.channel_id
),
lottery_daily AS (
    SELECT
        DATE(l.delivery_time) AS biz_date,
        l.channel_id,
        SUM(l.amount) AS lottery_amount
    FROM dwd_order_lottery l
    WHERE l.status = 2
      AND l.tenant_plat_id = :tenant_plat_id
      AND l.channel_id = :channel_id
      AND l.delivery_time >= :start_date
      AND l.delivery_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY DATE(l.delivery_time), l.channel_id
),
all_days AS (
    SELECT biz_date, channel_id FROM login_daily
    UNION
    SELECT biz_date, channel_id FROM register_daily
    UNION
    SELECT biz_date, channel_id FROM deposit_daily
    UNION
    SELECT biz_date, channel_id FROM first_deposit_split
    UNION
    SELECT biz_date, channel_id FROM withdraw_daily
    UNION
    SELECT biz_date, channel_id FROM bet_daily
    UNION
    SELECT biz_date, channel_id FROM rebate_daily
    UNION
    SELECT biz_date, channel_id FROM add_sub_daily
    UNION
    SELECT biz_date, channel_id FROM vip_award_daily
    UNION
    SELECT biz_date, channel_id FROM activity_daily
    UNION
    SELECT biz_date, channel_id FROM task_daily
    UNION
    SELECT biz_date, channel_id FROM promote_daily
    UNION
    SELECT biz_date, channel_id FROM lottery_daily
)
SELECT
    a.biz_date,
    a.channel_id,
    COALESCE(ld.login_user_count, 0) AS login_user_count,
    COALESCE(rd.register_user_count, 0) AS register_user_count,
    COALESCE(dd.deposit_user_count, 0) AS deposit_user_count,
    COALESCE(dd.deposit_amount, 0) AS deposit_amount,
    COALESCE(wd.withdrawal_user_count, 0) AS withdrawal_user_count,
    COALESCE(wd.withdrawal_amount, 0) AS withdrawal_amount,
    COALESCE(dd.deposit_amount, 0) - COALESCE(wd.withdrawal_amount, 0) AS charge_withdraw_diff,
    COALESCE(dd.first_deposit_user_count, 0) AS first_deposit_user_count,
    COALESCE(fd.new_customer_first_deposit_user_count, 0) AS new_customer_first_deposit_user_count,
    COALESCE(fd.develop_user_count, 0) AS develop_user_count,
    ROUND(COALESCE(dd.first_deposit_user_count, 0) / NULLIF(COALESCE(rd.register_user_count, 0), 0), 4) AS first_deposit_rate,
    COALESCE(dd.first_deposit_amount, 0) AS first_deposit_amount,
    ROUND(COALESCE(dd.first_deposit_amount, 0) / NULLIF(COALESCE(dd.first_deposit_user_count, 0), 0), 2) AS first_deposit_avg_amount,
    COALESCE(dd.new_customer_deposit_amount, 0) AS new_customer_deposit_amount,
    COALESCE(bd.bet_user_count, 0) AS bet_user_count,
    COALESCE(bd.valid_bet_amount, 0) AS valid_bet_amount,
    COALESCE(bd.win_loss_amount, 0) AS win_loss_amount,
    ROUND(COALESCE(bd.valid_bet_amount, 0) / NULLIF(COALESCE(dd.deposit_amount, 0), 0), 4) AS bet_deposit_ratio,
    COALESCE(rb.rebate_amount, 0) AS rebate_amount,
    COALESCE(asd.discount_adjust_amount, 0) AS discount_adjust_amount,
    COALESCE(vad.vip_award_amount, 0) AS vip_award_amount,
    COALESCE(acd.activity_amount, 0) AS activity_amount,
    COALESCE(td.task_amount, 0) AS task_amount,
    COALESCE(pd.promote_activity_amount, 0) AS promote_activity_amount,
    COALESCE(ldy.lottery_amount, 0) AS lottery_amount,
    COALESCE(asd.discount_adjust_amount, 0)
      + COALESCE(vad.vip_award_amount, 0)
      + COALESCE(acd.activity_amount, 0)
      + COALESCE(td.task_amount, 0)
      + COALESCE(pd.promote_activity_amount, 0)
      + COALESCE(ldy.lottery_amount, 0) AS promotion_total_amount
FROM all_days a
LEFT JOIN login_daily ld
       ON a.biz_date = ld.biz_date AND a.channel_id = ld.channel_id
LEFT JOIN register_daily rd
       ON a.biz_date = rd.biz_date AND a.channel_id = rd.channel_id
LEFT JOIN deposit_daily dd
       ON a.biz_date = dd.biz_date AND a.channel_id = dd.channel_id
LEFT JOIN first_deposit_split fd
       ON a.biz_date = fd.biz_date AND a.channel_id = fd.channel_id
LEFT JOIN withdraw_daily wd
       ON a.biz_date = wd.biz_date AND a.channel_id = wd.channel_id
LEFT JOIN bet_daily bd
       ON a.biz_date = bd.biz_date AND a.channel_id = bd.channel_id
LEFT JOIN rebate_daily rb
       ON a.biz_date = rb.biz_date AND a.channel_id = rb.channel_id
LEFT JOIN add_sub_daily asd
       ON a.biz_date = asd.biz_date AND a.channel_id = asd.channel_id
LEFT JOIN vip_award_daily vad
       ON a.biz_date = vad.biz_date AND a.channel_id = vad.channel_id
LEFT JOIN activity_daily acd
       ON a.biz_date = acd.biz_date AND a.channel_id = acd.channel_id
LEFT JOIN task_daily td
       ON a.biz_date = td.biz_date AND a.channel_id = td.channel_id
LEFT JOIN promote_daily pd
       ON a.biz_date = pd.biz_date AND a.channel_id = pd.channel_id
LEFT JOIN lottery_daily ldy
       ON a.biz_date = ldy.biz_date AND a.channel_id = ldy.channel_id
ORDER BY a.biz_date, a.channel_id;
```
